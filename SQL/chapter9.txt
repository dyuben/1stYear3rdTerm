create table student

(STUD_ID VARCHAR2(4)NOT NULL, LNAME VARCHAR2(25), FNAME VARCHAR2(25), GENDER VARCHAR(1), DATE_ENROLLED DATE, GROUPLEADER VARCHAR2(4), COURSE_ID VARCHAR(7)NOT NULL));

create table course

(COURSE_ID VARCHAR2(7)NOT NULL, COURSE_DESC(50), MENTOR_ID(4));

create table mentor

(MENTOR_ID NUMBER(4)NOT NULL, MENTORNAME VARCHAR(25)NOT NULL);
